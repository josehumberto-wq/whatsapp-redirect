  <h2>Abrindo WhatsApp...</h2>

  <button onclick="abrir()">Abrir WhatsApp</button>


  <script>

    function abrir() {

      // Pega o número da URL

      const params = new URLSearchParams(location.search);

      let phone = params.get('phone') || '';

      phone = phone.replace(/\D/g, ''); // limpa tudo que não for número

      

      if (!phone) {

        alert('Número não encontrado');

        return;

      }


      const url = 'https://web.whatsapp.com/send?phone=' + phone;

      window.open(url, 'whatsapp_window'); // reutiliza a aba

    }


    // Tenta abrir automaticamente

    setTimeout(abrir, 800);

  </script>
